
pipeline {
    agent any

    stages {
        stage('Checkout') {
            steps {
                echo 'SCM checkout done by Jenkins'
            }
        }

        stage('Build') {
            steps {
                sh 'mvn clean package'
            }
        }

        stage('Archive') {
            steps {
                archiveArtifacts artifacts: 'target/*.jar'
            }
        }
    }
}
